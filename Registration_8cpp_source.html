341 void Registration::icp (const PCNormalPtr &src
342  , const PCNormalPtr &tgt
343  , Eigen::Matrix4d &transform)
344 {
345  Eigen::Matrix4d final_transform (Eigen::Matrix4d::Identity ());
346  pcl::console::TicToc timer;
347  CorrespondencesPtr all_correspondences (new Correspondences),
348  good_correspondences (new Correspondences);
349  PCNormalPtr output (new PCNormal);
350 
351  *output = *src;
352 
353  int iterations = 0;
354  // Convergence criteria class stops the loop
355  pcl::registration::DefaultConvergenceCriteria<double> converged (iterations, transform, *good_correspondences);
356 
357  // ICP loop
358  do
359  {
360  // Find correspondences
361  timer.tic();
362  findCorrespondences (output, tgt, *all_correspondences);
363  //cout<<"Find Correspondances "<<": "<<timer.toc()<<" Miliseconds"<<endl;
364 
365  // Visualize first iteration all correspondances
366  // if(iterations==0)
367  // visualizeCorrespondances(output,tgt,output,tgt,all_correspondences);
368 // PCL_DEBUG ("Number of correspondences found: %d\n", all_correspondences->size ());
369 
370  if (rejection)
371  {
372  // Reject correspondences
373  timer.tic();
374  rejectBadCorrespondences (all_correspondences, output, tgt, *good_correspondences);
375  //cout<<"Reject Correspondances "<<": "<<timer.toc()<<" Miliseconds"<<endl;
376 // PCL_DEBUG ("Number of correspondences remaining after rejection: %d\n", good_correspondences->size ());
377  }
378  else
379  timer.tic();
380  *good_correspondences = *all_correspondences;
381 
382  // Visualize first iteration good correspondances
383  // if(iterations==0)
384  // visualizeCorrespondances(output,tgt,output,tgt,good_correspondences);
385 
386  // Find transformation
387  findTransformation (output, tgt, good_correspondences, transform);
388  //cout<<"Find Transformation "<<": "<<timer.toc()<<" Miliseconds"<<endl;
389  // Obtain the final transformation
390  final_transform = transform * final_transform;
391 
392  // Transform the data
393  timer.tic();
394  transformPointCloudWithNormals (*src, *output, final_transform.cast<float> ());
395  //cout<<"Transform Cloud "<<": "<<timer.toc()<<" Miliseconds"<<endl;
396  // Check if convergence has been reached
397  ++iterations;
398 
399  }
400  while (!converged);
401  cout<< "Took "<<iterations<<" to converge"<<endl;
402  // cout<<"Converged "<<converged<<endl;
403  // view (src, tgt, good_correspondences);
404  // view (output, tgt, good_correspondences);
405  transform = final_transform;
406 }
