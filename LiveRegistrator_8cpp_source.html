72 void LiveRegistrator::run ()
73 {
74 
75  PCXYZRGBPtr Viewer_result;
76  // Viewer initialization
77  initializeCamera(cloud_viewer_);
78 
79  // Senso initialization and start
80  initilizeSensor(grabber_);
81  grabber_->start();
82 
83  // Initialization of the registration object
84  Modelization::Registration registrator(true, true, 0.09, 1000);
85  PCXYZRGBPtr result;
86 
87  while (!cloud_viewer_->wasStopped ())
88  {
89  PCXYZRGBCPtr cloud; // input cloud
90 
91  // See if we can get a cloud if not try again
92  if (cloud_mutex_.try_lock ()){
93  cloud_.swap (cloud);
94  cloud_mutex_.unlock ();
95  }
96 
97  // Run if a cloud is available
98  if(cloud){
99  // Start the Registration process using cloud
100  if(registrator.addCloud(cloud)){
101  cout<<"Added Cloud";
102  }
103  // Finished Processing?
104  if(registrator.getResult(Viewer_result))
105  {
106  // Get the resulting image.
107  cout<<"ShowingResult"<<endl;
108  // // Create view or update resulting cloud view depending on if its the first cloud or not.
109  if (!cloud_viewer_->updatePointCloud(Viewer_result, "ResultCloud"))
110  {
111  //pcl::io::savePCDFileASCII("results/threadedresult.pcd",*Viewer_result);
112  cloud_viewer_->addPointCloud(Viewer_result, "ResultCloud",v1);
113  cout<<"ShowingResult first time"<<endl;
114  }
115  }
116  // Create view or update input cloud view depending on if its the first cloud or not. (Video Feed)
117  if (!cloud_viewer_->updatePointCloud (cloud, "OpenNICloud"))
118  {
119  cloud_viewer_->addPointCloud (cloud, "OpenNICloud",v2);
120  // cout<<"ShowingLive Feed"<<endl;
121  }
122  }
123 
124  // if no cloud is available do nothing placeholder for debug purposes.
125  else
126  {
127  // cout<<"No Image"<<endl;
128  continue;
129  }
130 
131  cloud_viewer_->spinOnce();
132 
133  }
134 
135  //When the window is closed stop the grabber and finish the function.
136  grabber_->stop ();
137  cloud_connection.disconnect ();
138  cout<<"The End"<<endl;
